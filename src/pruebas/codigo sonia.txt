textBox tbNum, tbBin, tbHex;

int nIn,nb,rb,db,nh,rh,dh;
string nBin, nbAux, nHex, nhAux;

def binario{
	LeeTextBox(tbNum, nIn);
	nb := nIn;
	rb := nIn;
	db := nIn;

	while(db > 0){
		db := nb / 2;
		rb := nb - (db * 2);
		nb := db;
		nbAux := nBin;
		
		switch(rb){
			case 0
			{ nBin := "0"; }
			break;
			case 1
			{ nBin := "1"; }
			break;
		}

		Concat(nBin, nbAux);
	}
	ImprimeTextBox(tbBin, nBin);
}

def hexadecimal{

		nhAux := nHex;

			break;

			{ nHex := "5"; }
			break;
			case 6
			{ nHex := "6"; }
			break;
			case 7
			{ nHex := "7"; }
			break;
			case 8
			{ nHex := "8"; }
			break;
			case 9
			{ nHex := "9"; }
			break;
			case 10
			{ nHex := "A"; }
			break;
			case 11
			{ nHex := "B"; }
			break;
			case 12
			{ nHex := "C"; }
			break;
			case 13
			{ nHex := "D"; }
			break;
			case 14
			{ nHex := "E"; }
			break;
			case 15
			{ nHex := "F"; }
			break;
		}
		Concat(nHex, nhAux);
	}
	ImprimeTextBox(tbHex, nHex);
}

def calcula{
	binario();
